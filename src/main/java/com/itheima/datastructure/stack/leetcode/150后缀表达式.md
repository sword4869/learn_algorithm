
- 注意`!t.equals("+")`而不是`t != "+"`，这是String
- 后弹者运算先弹者的顺序！
```java
class Solution {
    public int evalRPN(String[] tokens) {
        Stack<Integer> s = new Stack<>();
        for (String t : tokens) {
            if (!t.equals("+") && !t.equals("-") && !t.equals("*") && !t.equals("/")) {
                s.push(Integer.valueOf(t));
            } else {
                // 后弹者运算先弹者
                int b = s.pop();
                int a = s.pop();
                int c = -1;
                switch (t) {
                    case "+":
                        c = a + b;
                        break;
                    case "-":
                        c = a - b;
                        break;
                    case "*":
                        c = a * b;
                        break;
                    case "/":
                        c = a / b;
                        break;

                }
                s.push(c);
            }
        }
        return s.pop();
    }
}
```
