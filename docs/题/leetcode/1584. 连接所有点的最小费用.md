## Prim

```java
// 1584. 连接所有点的最小费用
class Solution {
    public static final int INF = Integer.MAX_VALUE >> 1;

    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        
        int[][] w = new int[n + 1][n + 1];
        for (int i = 1; i <= n; i++) {
            for (int j = i + 1; j <= n; j++) {
                int[] p1 = points[i - 1];
                int[] p2 = points[j - 1];
                w[i][j] = Math.abs(p1[0] - p2[0]) + Math.abs(p1[1] - p2[1]);
                w[j][i] = w[i][j];
            }
        }

        boolean[] vis = new boolean[n + 1];

        int[] dist = new int[n + 1];
        Arrays.fill(dist, INF);
        dist[1] = 0;

        int[] pre = new int[n + 1];     // 默认指向0节点

        prim(vis, dist, pre, w, n);

        int sum = 0;
        for (int i = 1; i <= n; i++) {
            sum += dist[i];
        }
        return sum;
    }

    public void prim(boolean[] vis, int[] dist, int[] pre, int[][] w, int n) {
        // 最小生成树只需N-1条边
        for (int p = 1; p < n; p++) {
            // 未遍历的点中找到距离点集最小的点
            int cur = 0;
            for (int i = 1; i <= n; i++) {
                if (!vis[i] && dist[i] < dist[cur]) {
                    cur = i;
                }
            }
            vis[cur] = true;

            // 未遍历的点中找到当前点的邻居节点，且当前点到邻居节点的边确实比原本最短边短
            for (int i = 1; i <= n; i++) {
                if (!vis[i] && w[cur][i] > 0 && w[cur][i] < dist[i]) {
                    // 则连接当前点和邻居点，其最短边是它，起点是当前点
                    dist[i] = w[cur][i];
                    pre[i] = cur;
                }
            }
        }
    }
}
```

## Kruskal

```java
class Solution {
    public int minCostConnectPoints(int[][] points) {
        int n = points.length;
        DisjointSet ds = new DisjointSet(n);

        PriorityQueue<int[]> edges = new PriorityQueue<>((a, b) -> a[2] - b[2]);
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int[] p1 = points[i];
                int[] p2 = points[j];
                int w = Math.abs(p1[0] - p2[0]) + Math.abs(p1[1] - p2[1]);
                edges.offer(new int[] { i, j, w });
            }
        }

        return kruskal(ds, edges, n);
    }

    public int kruskal(DisjointSet ds, PriorityQueue<int[]> edges, int n) {
        int sum = 0;
        for (int p = 0; p < n - 1;) {
            int[] edge = edges.poll();
            int i = edge[0], j = edge[1], w = edge[2];
            if (ds.join(i, j)) {
                sum += w;
                p++;
            }
        }
        return sum;
    }
}

class DisjointSet {
    int[] rank;
    int[] pre;

    public DisjointSet(int n) {
        rank = new int[n];
        pre = new int[n];
        for (int i = 0; i < n; i++) {
            pre[i] = i;
        }
    }

    public int find(int x) {
        if (x == pre[x]) {
            return x;
        } else {
            return pre[x] = find(pre[x]);
        }
    }

    public boolean join(int x, int y) {
        // 找到并集的根
        int px = find(x);
        int py = find(y);
        // 根相同，合并失败；根不同，比较层级，根的pre是另一个根
        if (px == py) {
            return false;
        }
        if (rank[px] == rank[py]) {
            rank[px]++;
        }
        if (rank[px] < rank[py]) {
            pre[px] = py;
        } else {
            pre[py] = px;
        }
        return true;
    }
}
```
