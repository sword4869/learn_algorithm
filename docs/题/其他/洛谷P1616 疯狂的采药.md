// https://www.luogu.com.cn/problem/P1616
import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int cap = sc.nextInt(), n = sc.nextInt();
        int[] w = new int[n];
        int[] v = new int[n];
        int i = 0;
        while (i < n) {
            w[i] = sc.nextInt();
            v[i] = sc.nextInt();
            i++;
        }
        Solution solution = new Solution();
        long res = solution.packInfinity(w, v, n, cap);
        System.out.println(res);

    }
}

class Solution {
    // n个物品，cap背包容量
    public long packInfinity(int[] w, int[] v, int n, int cap) {
        long[][] dp = new long[n][cap + 1];
        for(int c = 1; c<=cap;c++){
            if(w[0] <= c){
                dp[0][c] = dp[0][c-w[0]]+v[0];
            }
        }
        for (int i = 1; i < n; i++) {
            for (int c = 1; c <= cap; c++) {
                if (w[i] > c) {
                    dp[i][c] = dp[i-1][c];
                }
                else{
                    // 装的下：不选和选物品 i 这两种方案的较大值
                    dp[i][c] = Math.max(dp[i-1][c], dp[i][c - w[i]] + v[i]);
                }
            }
        }
        return dp[n-1][cap];
    }
}

/* 完全背包：动态规划 */
public long packInfinity(int[] w, int[] v, int n, int cap) {
    long[] dp = new long[cap + 1];
    // 列首是0，使用默认
    for (int i = 0; i < n; i++) {
        for (int c = 1; c <= cap; c++) {
            // 装不下就维持不变
            if (w[i] <= c) {
                // 装的下：不选和选物品 i 这两种方案的较大值
                dp[c] = Math.max(dp[c], dp[c - w[i]] + v[i]);
            }
        }
    }
    return dp[cap];
}
