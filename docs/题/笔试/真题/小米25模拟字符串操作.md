import java.util.*;

public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        String str = in.next();
        while(true){
            String str2 = delete(str);
            if(str.equals(str2)){
               System.out.println(str2);
                return;
            }
            str = str2;
        }
    }
    
    public static String delete(String str){
        char[] cs = str.toCharArray();
        int n = cs.length;
        if(n <= 1){
            return str;
        }
        boolean[] checked = new boolean[n];
        for(int i = 0; i < n; i++){
            if(i == 0){
                continue;
            }
            if(cs[i] == cs[i-1]){
                checked[i] = true;
                checked[i-1] = true;
                i++;
            }
        }
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < n; i++){
            if(checked[i] == false){
                sb.append(cs[i]);
            }
        }
        return sb.toString();
    }
}
