    public int[] sortArray(int[] nums) {
        int[] t = new int[nums.length];
        mergeSort(nums, 0, nums.length - 1, t);
        return nums;
    }

    public void mergeSort(int[] nums, int i, int j, int[] t) {
        if (i == j) {   // å½“å­æ•°ç»„é•¿åº¦ä¸º 1 æ—¶ç»ˆæ­¢é€’å½’
            return;

        mergeSort(nums, m + 1, j, t);

        merge(nums, i, m, m + 1, j, t);
        System.arraycopy(t, i, nums, i, j - i + 1);
    }

    public void merge(int[] nums, int mi, int mj, int ni, int nj, int[] t) {
        int k = mi;
        while (mi <= mj && ni <= nj) {
            if (nums[mi] < nums[ni]) {
                t[k] = nums[mi];
                mi++;
            } else {
                t[k] = nums[ni];
                ni++;
            }
            k++;
        }
        if (mi <= mj) {
            System.arraycopy(nums, mi, t, k, mj - mi + 1);
        } else {
            System.arraycopy(nums, ni, t, k, nj - ni + 1);
        }
    }
}

class Solution {
    public int[] sortArray(int[] nums) {
        int[] t_arr = new int[nums.length];
        split(nums, 0, nums.length - 1, t_arr);
        return nums;
    }

    public void insertion(int[] nums, int left, int right){
        for(int low = left + 1; low <= right; low++){
            int t = nums[low];
            int i = low - 1;
            while(i >= left && nums[i] > t){
                nums[i + 1] = nums[i];
                i--;
            }
            if(i != low -1){
                nums[i + 1] = t;
            }
        }
    }

    public void split(int[] nums, int left, int right, int[] t_arr) {
        if (left == right) {
            return;
        }

            return;
