次次大

```java
int first = -1, second = -1, third = -1;
for(int i=0;i<n;i++){
    int num = nums[i];
    if(num > first){
        third = second;
        second = first;
        first = num;
    }
    // num < first && num > second
    else if (num > second){
        third = second;
        second = num;
    }
    // num < first && num < second && num > third
    else if (num > third){
        third = num;
    }
}
```
